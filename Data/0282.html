We are looking to convert our NCEP model netCDF files to use CF.

Follows is a list of parameters in those models not apparently covered
by CF standard names.

I either indicate "name derived from grib" to indicate a possible
addition, or "use" which states an existing one which is probably? ok to
use, so that a cross reference to the GRIB id can be added.

disclaimer: i am not a meteorologist, so if anyone knows an existing
mapping or beterr name, i certainly wont mind.


 From ETA model

float helc_lfhg(record, lfhg, y, x);
    :long_name = "Storm relative helicity at layer between 2 height
levels above ground";
    :units = "m2/s2";
    :GRIB_parameter_number = 190;

name derived from grib = storm_relative_helicity

   float cape_lpdg(record, lpdg, y, x);
    :long_name = "Convective available potential energy at layer between
levels at specif. pressure diffs from ground";
    :units = "J/kg";
    :GRIB_parameter_number = 157;

name derived from grib = convective_available_potential_energy


   float cin_lpdg(record, lpdg, y, x);
    :long_name = "Convective inhibition at layer between levels at
specif. pressure diffs from ground";
    :units = "J/kg";
    :GRIB_parameter_number = 156;

name derived from grib = convective_inhibition


   float LI4_lpdg(record, lpdg, y, x);
    :long_name = "Best (4 layer) lifted index at layer between levels at
specif. pressure diffs from ground";
    :units = "degK";
    :GRIB_parameter_number = 132;

name derived from grib = best_4layer_lifted_index


   float pli_lpdg(record, lpdg, y, x);
    :long_name = "Parcel lifted index (to 500 hPa) at layer between
levels at specif. pressure diffs from ground";
    :units = "K";
    :GRIB_parameter_number = 24;

name derived from grib = parcel_lifted_index


float Psl_et(record, y, x);
    :long_name = "Mean sea level pressure (ETA model reduction) at mean
sea level";
    :units = "Pa";
    :GRIB_parameter_number = 130;

use = air_pressure_at_sea_level
